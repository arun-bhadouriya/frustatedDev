Linked List

A linked list is a linear data structure, in which the elements are not stored at contiguous memory locations. The elements in a linked list are linked using pointers.In simple words, a linked list consists of nodes where each node contains a data field and a reference(link) to the next node in the list.

Basics fundamentals action of a Linked lists are -

  1. INSERTION

      1. At the front of the linked list
      2. After a given node.
      3. At the end of the linked list.
  2. DELETION
    • Deletion At the head
    • Deletion at a given position
    • Deletion At tail
  3. SEARCHING
    • linear search( iterative approach)

        Node*temp=head;

            l++;
            temp=temp->next;

    void insertAtTail(Node*&head,int data){
        if(head==NULL){
            insertAtHead(head,data);
            return;
        }
    
        Node*temp=head;
        while(temp->next!=NULL){
            temp=temp->next;
        }
        Node*n=new Node(data);
        temp->next=n;
    }
    
    void insertInMiddle(Node*head , int data, int pos){
        if(head==NULL or pos==0){
            insertAtHead(head,data);
            return;
        }
    
        else if(pos>length(head)){
            insertAtTail(head,data);
            return;
        }
    
        int jump=1;
        Node*temp=head;
        while(jump < pos-1){
            temp=temp->next;
            jump++;
        }
        Node*n=new Node(data);
        n->next=temp->next;
        temp->next=n;
    }

    void reverse_LL(Node*&head){

        Node*C=head;
        Node*prev=NULL;
        Node*N=head;

        while(C!=NULL){
            //save the next node before altering the address of the current node
            N=C->next;
            //now change the current node adress to the previous node (=NULL)
            C->next=prev;

            //now move prev node pointing to current node and current node pointing to next node
            prev=C;
            C=N;
        }

        head=prev;
    }

    Node* MidPoint(Node*head){
        if(head==NULL or head->next==NULL) return head;
        Node*fastHorse=head->next;
        Node*slowHorse=head;

        while(fastHorse!=NULL and fastHorse->next!=NULL){
            fastHorse=fastHorse->next->next;
            slowHorse=slowHorse->next;   
        }
        return slowHorse;
    }

    node* kthNodeFromEnd(node*head, int k){
        node*slow=head;
        node*fast=head;

        while(fast!=NULL){
            if(k>0){
                fast=fast->next;
                k--;
            }
            else{
                fast=fast->next;
                slow=slow->next;
            }
        }
        return slow;
    }

    Node* merge(Node*a, Node*b){
        if(a==NULL) return b;

        if(b==NULL) return a;

        Node*c;

        if(a->data < b->data){
            c=a;
           c->next= merge(a->next,b);
        }
        else{
            c=b;
           c->next= merge(a,b->next);
        }

        return c;
    }

node* mergeSort(node*head){
    if(head==NULL or head->next==NULL) return head;

    //1. break by midpoint;
    node* midpoint = MidPoint(head);
    node*a = head;
    node*b= midpoint->next;

    midpoint->next=NULL;

    //sort
    a=mergeSort(a);
    b=mergeSort(b);

    //merge
    node*c = merge(a,b);
    return c;
}
                        
bool detectCycle(node*head){
    node*slow=head;
    node*fast=head;

    while(fast->next!=NULL and fast!=NULL){
        fast=fast->next->next;
        slow=slow->next;

        if(slow==fast){
            return true;
        }
    }

    return false;
}

    void push_front(dll*&head,dll*&tail,int d){
        if(head==NULL){
            head= new dll(d);
            tail=head;
            return;
        }
        dll*n=new dll(d);
        n->next=head;
        head->prev=n;
        head=n;
        // cout<prev<<endl;
    }
    
    //PUSH AT THE BACK OF THE DOUBLY LINKED LIST
    void push_back(dll*&head,dll*&tail,int d){
        if(head==NULL or tail==NULL) {
            push_front(head,tail,d);
            return;
        }
    
        dll*n=new dll(d);
        tail->next=n;
        n->prev=tail;
        tail=n;
    }
    //DELETE IN THE FRONT OF THE DOUBLY LINKED LIST
    void delete_front(dll*&head){
        if(head==NULL){
            return;
        }
    
        dll*temp = head->next;
        delete head;
        head=temp;
        temp->prev=NULL;
    }
    
    //DELETE IN THE BACK OF THE DOUBLY LINKED LIST
    void delete_back(dll*&tail){
        if(tail==NULL){
            return;
        }
        dll*temp=tail->prev;
        delete tail;
        temp->next=NULL;
        tail = temp;
    }
