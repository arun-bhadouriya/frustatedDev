Linked Lists

A linked list is a linear data structure, in which the elements are not stored at contiguous memory locations. The elements in a linked list are linked using pointers.In simple words, a linked list consists of nodes where each node contains a data field and a reference(link) to the next node in the list.

  1. INSERTION

  2. DELETION
    • Deletion At the head
    • Deletion at a given position
    • Deletion At tail
  3. SEARCHING
    • linear search( iterative approach)

    void insertAtHead( Node*&head, int data ){
        if(head==NULL){
            head=new Node(data);
            return;
        }

        Node * n= new Node(data);
        n->next = head;
        head = n;
    }
    int length(Node*head){
        Node*temp=head;
        int l=0;
        while(temp!=NULL){
            l++;
            temp=temp->next;
        }
        return l;
    }
    
    void insertAtTail(Node*&head,int data){
        if(head==NULL){
            insertAtHead(head,data);
            return;
        }
    
        Node*temp=head;
        while(temp->next!=NULL){
            temp=temp->next;
        }
        Node*n=new Node(data);
        temp->next=n;
    }
    
    void insertInMiddle(Node*head , int data, int pos){
        if(head==NULL or pos==0){
            insertAtHead(head,data);
            return;
        }
    
        else if(pos>length(head)){
            insertAtTail(head,data);
            return;
        }
    
        int jump=1;
        Node*temp=head;
        while(jump < pos-1){
            temp=temp->next;
            jump++;
        }
        Node*n=new Node(data);
        n->next=temp->next;
        temp->next=n;
    }

        //deletion
    
        //function to delete the head of the linked list

        void deleteHead(Node*&head){
            if(head==NULL){
                return;
            }
            Node* temp= head->next;
            delete head;
            head=temp;
        }

        // Function to delete the tail of the linked list

        void deleteTail(Node*&head){
            if(head==NULL) return;

            Node*temp=head;
            while(temp->next->next!=NULL){
                temp=temp->next;
            }
            Node*previous=temp;
            temp=temp->next;
            delete temp;
            previous->next=NULL;
        }

        // Function to delete at a given positon in a linked list;
        
        void deleteAtGivenPosition(Node*&head , int p ){
            if(head==NULL) return;

            else if(p==length(head)){
                deleteTail(head);
                return;
            }

            int jump=1;
            Node*temp=head;
            while(jump<=p-2){
                temp=temp->next;
                jump++;
            }

            Node*n= temp->next;
            temp->next=n->next;
            delete n;
        }

    //Iterative Approach
    void reverse_LL(Node*&head){

        Node*C=head;
        Node*prev=NULL;
        Node*N=head;

        while(C!=NULL){
            //save the next node before altering the address of the current node
            N=C->next;
            //now change the current node adress to the previous node (=NULL)
            C->next=prev;

            //now move prev node pointing to current node and current node pointing to next node< /span>
            prev=C;
            C=N;
        }

        head=prev;
    }
