Linked Lists

A linked list is a linear data structure, in which the elements are not stored at contiguous memory locations. The elements in a linked list are linked using pointers.In simple words, a linked list consists of nodes where each node contains a data field and a reference(link) to the next node in the list.

Node Structure is 

    class Node {
        public:
            int data;
            Node * next;
    };

A simple code snippet to create linked list (manuaally giving values)

    #include <iostream>

    //Insertion

    #include<iostream>
    using namespace std;

    class Node{
        public:
            int data;
            Node*next;

            Node(int d){
                data=d;
                next=NULL;
            }
    };

    void insertAtHead( Node*&head, int data ){
        if(head==NULL){
            head=new Node(data);
            return;
        }

        Node * n= new Node(data);
        n->next = head;
        head = n;
    }
    int length(Node*head){
        Node*temp=head;
        int l=0;
        while(temp!=NULL){
            l++;
            temp=temp->next;
        }
        return l;
    }
    
    void insertAtTail(Node*&head,int data){
        if(head==NULL){
            insertAtHead(head,data);
            return;
        }
    
        Node*temp=head;
        while(temp->next!=NULL){
            temp=temp->next;
        }
        Node*n=new Node(data);
        temp->next=n;
    }
    
    void insertInMiddle(Node*head , int data, int pos){
        if(head==NULL or pos==0){
            insertAtHead(head,data);
            return;
        }
    
        else if(pos>length(head)){
            insertAtTail(head,data);
            return;
        }
    
        int jump=1;
        Node*temp=head;
        while(jump < pos-1){
            temp=temp->next;
            jump++;
        }
        Node*n=new Node(data);
        n->next=temp->next;
        temp->next=n;
    }

    void print_LL(Node*head){
        Node*temp=head;
        while(temp!=NULL){
            cout<data<<"->";
            temp=temp->next;
        }
    }

    int main(){
        Node* head =NULL;
        insertAtHead(head,1);
        insertAtHead(head,2);
        insertAtHead(head,3);
    
        print_LL(head); //this will OUTPUT: 3->2->1-> 
        insertInMiddle(head,4,2);
        cout<<endl;
        print_LL(head);  //this will OUTPUT: 3->4->2->1->
        cout<<endl;
        insertAtTail(head,5);
        print_LL(head); //this will OUTPUT: 3->4->2->1->5-> 
    }

    //deletion
    
    //function to delete the head of the linked list

    void deleteHead(Node*&head){
        if(head==NULL){
            return;
        }
        Node* temp= head->next;
        delete head;
        head=temp;
    }

    // Function to delete the tail of the linked list

    void deleteTail(Node*&head){
        if(head==NULL) return;

        Node*temp=head;
        while(temp->next->next!=NULL){
            temp=temp->next;
        }
        Node*previous=temp;
        temp=temp->next;
        delete temp;
        previous->next=NULL;
    }

    // Function to delete at a given positon in a linked list;
    
    void deleteAtGivenPosition(Node*&head , int p ){
        if(head==NULL) return;

        else if(p==length(head)){
            deleteTail(head);
            return;
        }

        int jump=1;
        Node*temp=head;
        while(jump<=p-2){
            temp=temp->next;
            jump++;
        }

        Node*n= temp->next;
        temp->next=n->next;
        delete n;
    }
