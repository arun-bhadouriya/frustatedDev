#include <iostream>
#include<vector>
using namespace std;
class stack{
    private: 
        vector <int> v;
        int top=-1;
    public:
        bool empty(){
            return v.empty();
        }
        void push(int d){
            v.push_back(d);
            top=d;
        }
        void pop(){
            if(!empty()){
                v.pop_back();
                top=v.back();
            }
        }
        
        void getTop(){
            if(top!=NULL){
                cout<<top<<endl;
            } 
            else{
                cout<<-1<<endl;
            }
        }
};

    template<typename T,typename U>

    private:
        vector<T> v;

        }
        bool empty(){
            return v.size()==0;
        }
        void pop(){
            if(!empty()){
                    v.pop_back();
            }
        }
        T top(){
            return v[v.size()-1];
        }

                        //Q. reverse a stack using another stack ?
                        
                        
                        void transfer(stack<int> &s1, stack <int> &s2,int n){
                        
                            for(int i=0;i<n;i++){
                                s2.push(s1.top());
                                s1.pop();
                            }
                        }
                        
                        void reverse(stack<int> &a){
                            stack <int>s;
                            int n = a.size();
                            for(int i=0; i<n; i++){
                                
                                int x = a.top();
                                a.pop();
                        
                                
                                transfer(a,s,n-i-1);
                                a.push(x);
                                transfer(s,a,n-i-1);
                                
                            }
                        }

                        void insertAtBottom(stack<int> &s,int x){
                            if(s.empty()){
                                s.push(x);
                                return;
                            }
                            int data = s.top();
                            s.pop();
                            insertAtBottom(s,x);
                            s.push(data);
                        }
                        
                        void recursiveReverse(stack<int> &s){
                            //base case 
                            if(s.empty()){
                                return;
                            }
                        
                            int x = s.top();
                            s.pop();
                            recursiveReverse(s);
                            insertAtBottom(s,x);
                        }
